artistice *add_film_artistic(artistice *prim) {
	artistice *temp,*nou;
	char titlu[22], regizor[22],nume_pers_imprumut[22];
	int an_ap, durata, disp;
	data *imprumut,*returnare;
	system("cls");

	printf ("Introduceti date\n");
	printf("Titlu: ");
	fflush(stdin);
	gets(titlu);
	printf("Regizor: ");
	fflush(stdin);
	gets(regizor);
	printf("An aparitie: ");
	scanf("%d",&an_ap);
	printf("Durata: ");
	scanf("%d",&durata);
	printf("Disponibilitate  (0 - NU, 1 - DA):");
	scanf("%d",&disp);
	if(!disp) {
		imprumut=(data*)malloc(sizeof(data));
		returnare=(data*)malloc(sizeof(data));
		printf("Nume persoana imprumut:");
		fflush(stdin);
		gets(nume_pers_imprumut);
		printf("Data imprumut\nZi: ");
		scanf("%d",&imprumut->zi);
		printf("Luna: ");
		scanf("%d",&imprumut->luna);
		printf("An: ");
		scanf("%d",&imprumut->an);
		
		printf("Data returnat\nZi: ");
		scanf("%d",&returnare->zi);
		printf("Luna: ");
		scanf("%d",&returnare->luna);
		printf("An: ");
		scanf("%d",&returnare->an);
		
	}
	
	nou=(artistice*)malloc(sizeof(artistice));
	nou->titlu=(char*)malloc(sizeof(titlu));
	nou->regizor=(char*)malloc(sizeof(regizor));
	if(!disp) {
		nou->nume_pers_imprumut=(char*)malloc(sizeof(nume_pers_imprumut));
		strcpy(nou->nume_pers_imprumut,nume_pers_imprumut);
		nou->imprumut=(data*)malloc(sizeof(data));
		nou->returnare=(data*)malloc(sizeof(data));
		
		nou->imprumut->an=imprumut->an;
		nou->imprumut->luna=imprumut->luna;
		nou->imprumut->zi=imprumut->zi;		
		
		nou->returnare->an=returnare->an;
		nou->returnare->luna=returnare->luna;
		nou->returnare->zi=returnare->zi;
	}
	
	strcpy(nou->titlu,titlu);
	strcpy(nou->regizor,regizor);
	nou->an_aparitie=an_ap;
	nou->durata=durata;
	nou->disponibilitate=disp;
	
	if(prim==NULL) {
		prim=nou;
		prim->urm=NULL;
	}
	else {
		temp=prim;
		while(temp->urm!=NULL) {
			temp=temp->urm;
		}
		temp->urm=nou;
		nou->urm=NULL;
		
	}
	
	return prim;
}
