#include <stdio.h>

typedef enum disponibil{neimprumutat, imprumutat} disp;

typedef struct filme_artistice {

artistice add_filme_artistice(artistice prim) {
	artistice nou, temp;
	char titlu[22], regizor[22],nume_persoana_imprumut[22];
	int an_aparitie, durata,disponibilitate;
	data imprumutat, returnare;
	
	system("cls");            
	
	printf("\nIntroduceti datele ");
	printf("\ntitlu: ");
	fflush(stdin);
	gets(titlu);
	printf("regizor: ");
	fflush(stdin);
	gets(regizor);             
	printf("an apritie: ");
	scanf("%d",&an_aparitie);
	printf("durata: ");
	scanf("%d",&durata);
	printf("Disponibilitate:\t0-Neiprumutat\t\\\t1-Imprumutat:_____\b\b\b");
	scanf("%d",&disponibilitate);
	
	if(disponibilitate==1) {
		imprumutat=(data)malloc(sizeof(data));
		returnare=(data)malloc(sizeof(data));
		
		printf("nume persoana imprumut: ");
		fflush(stdin);
		gets(nume_persoana_imprumut);
		printf("data imprumutare\nzi: ");
		scanf("%d",&imprumutat->zi);
		printf("luna: ");
		scanf("%d",&imprumutat->luna);
		printf("an: ");
		scanf("%d",&imprumutat->an);
		
		printf("data returnare\nzi: ");
		scanf("%d",&returnare->zi);
		printf("luna: ");
		scanf("%d",&returnare->luna);
		printf("an: ");
		scanf("%d",&returnare->an);
	}
					 
	nou=(artistice)malloc(sizeof(artistice));
	nou->titlu=(char*)malloc(sizeof(titlu));
	nou->regizor=(char*)malloc(sizeof(regizor)+1);
	
	strcpy(nou->titlu,titlu);
	strcpy(nou->regizor,regizor);
	nou->an_aparitie=an_aparitie;
	nou->durata=durata;
	nou->disponibilitate=disponibilitate;
	
	if(nou->disponibilitate==1) {
		nou->nume_persoana_imprumut=(char*)malloc(sizeof(nume_persoana_imprumut));
		nou->imprumutat=(data)malloc(sizeof(data));
		nou->returnare=(data)malloc(sizeof(data));
		
		strcpy(nou->nume_persoana_imprumut,nume_persoana_imprumut);
		
		nou->imprumutat->zi=imprumutat->zi;
		nou->imprumutat->luna=imprumutat->luna;
		nou->imprumutat->an=imprumutat->an;
		
		nou->returnare->zi=returnare->zi;
		nou->returnare->luna=returnare->luna;
		nou->returnare->an=returnare->an;
	}
						  
	if(prim==NULL) {     
		prim=nou;
		prim->urm=NULL;      
	}                    
	else {
		temp=prim;
		while(temp->urm!=NULL) {
			temp=temp->urm;
		}                
		temp->urm=nou;
		nou->urm=NULL;
	}
return prim;  
}
